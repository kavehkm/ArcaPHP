Install apache2:

  1. sudo apt update
  2. sudo apt install apache2

Config firewall:

  1. sudo ufw app list
  2. sudo uf2 allow "Apache Full"

Install MySQL:

  1. sudo apt install mysql-server
  2. sudo mysql_secure_installation

Install PHP:

  1. sudo apt install php libapache2-mod-php
  2. sudo apt install php-mysql php-curl php-mbstring php-gd php-zip ...

VirtualHost:

  1. sudo mkdir /var/www/arca/
  2. sudo chown -R $USER:$USER /var/www/arca
  3. sudo chmod -R 755 /var/www/arca
  4. touch /var/www/arca/index.php
  5. echo "<?php echo 'Hello world'; ?>" > /var/www/arca/index.php>
  6. touch /etc/apache2/site-available/arca.conf: 
        <VirtualHost *:80>
        ServerAdmin admin@arca.net

        ServerName arca.net
        ServerAlias www.arca.net

        DocumentRoot /var/www/arca

        ErrorLog ${APACHE_LOG_DIR}/error.log
        CustomLog ${APACHE_LOG_DIR}/access.log combined
    </VirtualHost>
  7. sudo a2ensite arca.conf
  8. sudo echo "127.0.0.1 arca.net" >> /etc/hosts
  9. sudo systemctl restart apache2
